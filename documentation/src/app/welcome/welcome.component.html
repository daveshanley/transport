Step 3: Hello World

// listen for requests on 'my-channel' and return a response.
this.bus.respondOnce('my-channel')
    .generate(
        (request: string) => {
            this.log.info(`Request Received: ${request}, Sending Response...`);
            return 'world';
        }
    );

// send request 'hello' on channel 'my-channel'.
this.bus.requestOnce('my-channel', 'hello')
    .handle(
        (response: string) => {
            this.log.info(`hello ${response}`);
        }
    );
